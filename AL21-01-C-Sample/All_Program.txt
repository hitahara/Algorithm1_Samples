#include <stdio.h>

int gcd(int m, int n)
{
  int r;

  do
  {
    r = m % n;
    m = n; n = r;
  } while(r != 0);
  
  int gcd = m;
  
  return gcd;
}

int main()
{
  int m, n, gcd_answer;
  
  printf("Type in m...\n");
  scanf("%d", &m);
  printf("Type in n...\n");
  scanf("%d", &n);
  
  gcd_answer = gcd(m, n);
  
  printf("GCD: %d.\n", gcd_answer);
  
  return 0;
}

#include <stdio.h>

int gcd(int m, int n)
{
  if(n == 0)
  {
    return m;
  } 
  else 
  {
    return gcd(n, m % n);
  }
}

int main()
{
  int m, n, gcd_answer;
  
  printf("Type in m...\n");
  scanf("%d", &m);
  printf("Type in n...\n");
  scanf("%d", &n);
  
  gcd_answer = gcd(m, n);
  
  printf("GCD: %d.\n", gcd_answer);
  
  return 0;
}
